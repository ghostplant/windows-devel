画笔颜色：

CPen Pen;

Pen.CreatePen(WS_SOLID,1,RGB(x,y,z));

CClientDC/CDC/CPaintDC *dc(this);

dc->SelectObject(&Pen);

Pen.DeleteObject();

//然后使用 dc 变量来绘图

字体：

CFont Font;

Font.CreateFont(20/*字号*/,0/*加宽*/,0/*倾斜度*/,0,0/*粗度*/,0/*斜体*/,0/*下划线*/,0/*删除线*/,ANSI_CHARSET/*编码*/,OUT_DEFAULT_PRECIS,CLIP_DEFAULT_PRECIS,DEFAULT_QUALITY,DEFAULT_PITCH,"宋体");

CClientDC/CDC/CPaintDC *dc(this);

dc->SelectObject(&Font);

Font.DeleteObject();

//然后使用 dc 变量来输出文字

显示位图：

CPaintDC dc(this);

CBitmap *pBmp=new CBitmap();

pBmp->LoadBitmap(IDB_BMPNAME);

// 载入位图

CDC *pDDC=new CDC();

pDDC->CreateCompatibleDC(&dc);

pDDC->SelectObject(pBmp);

// 场景准备

BITMAP bm;

pBmp->GetObject(sizeof(bm),&bm);

// 获取位图长宽

dc.StretchBlt(0,0,bm.bmWidth,bm.bmHeight,pDDC,0,0,bm.bmWidth,bm.bmHeight,SRCCOPY);

// 画图
